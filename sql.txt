CREATE TABLE EMPLOYEE (
ID INT IDENTITY NOT NULL PRIMARY KEY,
NAME VARCHAR(100) NOT NULL,
SUPERVISOR INT NULL,
BEGIN_DATE DATETIME NULL,
END_DATE DATETIME NULL,
CONSTRAINT FK_SUPERVISOR FOREIGN KEY SUPERVISOR REFERENCES EMPLOYEE(ID)
)

1. всех сотрудников, работающих в определенную дату:
SELECT NAME FROM EMPLOYEE WHERE BEGIN_DATE <= date AND END_DATE > date;
2. всех сотрудников, работающих после определенной даты
SELECT NAME FROM EMPLOYEE WHERE END_DATE > date;
3. данные о сотруднике с именем его руководителя
SELECT NAME, SUPERVISOR_NAME FROM EMPLOYEE, FK_SUPERVISOR WHERE FK_SUPERVISOR.SUPERVISOR_ID = EMPLOYEE.SUPERVISOR
